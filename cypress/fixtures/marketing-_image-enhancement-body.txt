  if (!shell) return;

  const video = shell.querySelector('video');
  if (!video) return;

  let revealed = false;
  const reveal = () => {
    if (revealed || shell.classList.contains('has-video-error')) return;
    revealed = true;
    shell.classList.add('is-video-ready');
  };

  const tryPlay = () => {
    if (video.dataset.autoplay !== 'true') {
      if (video.readyState >= 1 || video.currentSrc) reveal();
      return;
    }

    video.muted = true;
    video.autoplay = true;
    video.playsInline = true;

    const playPromise = video.play();
    if (playPromise && typeof playPromise.then === 'function') {
      playPromise.then(reveal).catch(() => {
        if (video.readyState >= 2 || video.currentSrc) reveal();
      });
    } else if (video.readyState >= 1 || video.currentSrc) {
      reveal();
    }
  };

  const loadVideo = () => {
    if (video.dataset.loaded === 'true') return;
    video.dataset.loaded = 'true';

    video.addEventListener('playing', reveal, { once: true });
    video.addEventListener('timeupdate', reveal, { once: true });
    video.addEventListener('canplay', tryPlay, { once: true });
    video.addEventListener('loadeddata', tryPlay, { once: true });
    video.addEventListener('loadedmetadata', tryPlay, { once: true });
    video.addEventListener('error', () => {
      shell.classList.add('has-video-error');
    }, { once: true });

    video.querySelectorAll('source[data-src]').forEach((source) => {
      const nextSrc = source.dataset.src;
      if (!nextSrc) return;
      source.src = nextSrc;
      source.removeAttribute('data-src');
    });

    video.load();
    tryPlay();

    window.setTimeout(() => {
      if (!revealed && !shell.classList.contains('has-video-error') && (video.readyState >= 1 || video.currentSrc)) {
        tryPlay();
        reveal();
      }
    }, 2200);
  };

  const scheduleLoad = () => window.requestAnimationFrame(loadVideo);
  if (document.readyState === 'complete') {
